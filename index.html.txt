    <script>
        // আপনার দেওয়া URL এখানে সেট করা হলো
        const SCRIPT_URL = "https://script.google.com/macros/s/AKfycbzE8YwMHm-3HiCWZT2g55nH24C62rGjJu-bCJiXkZi8yQKU99sRUJDEKdTK_-4_PKwXbA/exec";

        function toggleForm(type) {
            document.getElementById('loginSection').classList.toggle('hidden', type === 'register');
            document.getElementById('registerSection').classList.toggle('hidden', type === 'login');
            document.getElementById('msgBox').classList.add('hidden');
        }

        function showMessage(msg, type) {
            const box = document.getElementById('msgBox');
            box.classList.remove('hidden', 'bg-red-100', 'text-red-700', 'bg-green-100', 'text-green-700', 'bg-blue-100', 'text-blue-700', 'bg-yellow-100', 'text-yellow-800');
            box.innerText = msg;
            
            if(type === 'error') box.classList.add('bg-red-100', 'text-red-700');
            else if(type === 'success') box.classList.add('bg-green-100', 'text-green-700');
            else if(type === 'wait') box.classList.add('bg-yellow-100', 'text-yellow-800', 'border', 'border-yellow-300');
            else box.classList.add('bg-blue-100', 'text-blue-700');
            
            box.classList.remove('hidden');
        }

        async function handleRegister() {
            const data = {
                action: "register",
                name: document.getElementById('regName').value,
                email: document.getElementById('regEmail').value,
                mobile: document.getElementById('regMobile').value,
                shop: document.getElementById('regShop').value,
                txid: document.getElementById('regTxid').value,
                password: document.getElementById('regPass').value
            };

            if(!data.name || !data.mobile || !data.txid || !data.password) {
                alert("দয়া করে সব তথ্য পূরণ করুন!"); return;
            }

            showMessage("তথ্য পাঠানো হচ্ছে, অপেক্ষা করুন...", "info");
            
            try {
                const response = await fetch(SCRIPT_URL, {
                    method: 'POST',
                    mode: 'no-cors', // Google Apps Script এর জন্য প্রয়োজন
                    body: JSON.stringify(data)
                });
                // যেহেতু no-cors, আমরা সরাসরি JSON রেসপন্স পাবো না, তাই একটু ট্রিক করতে হবে
                showMessage("আপনার অ্যাকাউন্ট এর তথ্য আমাদের কাছে জমা হয়েছে, দয়া করে অপেক্ষা করে যাবেন আমাদের টিম এটি যাচাই করে আপনার আইডি Accept করবেন।", "wait");
            } catch (e) {
                showMessage("সাবমিট হয়েছে। অনুগ্রহ করে কিছু সময় পর লগইন করার চেষ্টা করুন।", "wait");
            }
        }

        async function handleLogin() {
            const mobile = document.getElementById('loginMobile').value;
            const pass = document.getElementById('loginPassword').value;

            if(!mobile || !pass) {
                alert("মোবাইল নম্বর ও পাসওয়ার্ড দিন"); return;
            }

            showMessage("যাচাই করা হচ্ছে...", "info");

            // লগইন এর জন্য আমরা Fetch API ব্যবহার করবো
            fetch(SCRIPT_URL + "?action=login&mobile=" + mobile + "&pass=" + pass)
            .then(res => res.json())
            .then(result => {
                if(result.result === "success") {
                    if(result.status === "Pending") {
                        showMessage("আপনার অ্যাকাউন্ট এর তথ্য আমাদের কাছে জমা হয়েছে, দয়া করে অপেক্ষা করে যাবেন আমাদের টিম এটি যাচাই করে আপনার আইডি Accept করবেন।", "wait");
                    } else {
                        showMessage("লগইন সফল! প্রবেশ করা হচ্ছে...", "success");
                        localStorage.setItem("userMobile", mobile);
                        setTimeout(() => { window.location.href = "dashboard.html"; }, 1500);
                    }
                } else {
                    showMessage("ভুল মোবাইল নম্বর অথবা পাসওয়ার্ড!", "error");
                }
            })
            .catch(err => {
                // যদি Get Request এ সমস্যা হয় তবে সরাসরি Apps Script চেক করা প্রয়োজন
                showMessage("সার্ভারে সমস্যা হচ্ছে। দয়া করে অ্যাডমিনের সাথে যোগাযোগ করুন।", "error");
            });
        }
    </script>
